const int maxn = 1003;
int n, m;
int fa[maxn];
bool ansRes[maxn];

int find(int x) {
	return x == fa[x] ? x : fa[x] = find(fa[x]);
}

int main() {
	while (cin >> n && n) {
		for (int i = 0; i <= n; i++) {
			fa[i] = i;
		}
		memset(ansRes, false, sizeof(ansRes));

		cin >> m;
		int u, v;
		for (int i = 0; i < m; i++) {
			cin >> u >> v;
			u = find(u);
			v = find(v);
			if (u != v) {
				fa[u] = v;
			}
		}
		int ans = 0;

		for (int i = 1; i <= n; i++) {
			if (!ansRes[ find(i)]) {
				ans++;
				ansRes[find(i)] = true;
			}
		}
		cout << --ans << endl;
	}

	return 0;
}
