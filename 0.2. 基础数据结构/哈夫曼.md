例3.3 输出叶节点的权值与值的乘积的和的最小值（哈夫曼）
```C++
const int maxn = 1003;
priority_queue<int, vector<int>, greater<int> > pq;
int main() {
	int n,tmpi;
	while (cin >> n) {
		for (int i = 0; i < n; i++) {
			cin >> tmpi;
			pq.push(tmpi);
		}
		int ans = 0, n1, n2;
		while (pq.size()>=2) {
			n1 = pq.top();
			pq.pop();
			n2 = pq.top();
			pq.pop();
			ans += n1;
			ans += n2;
			pq.push(n1 + n2);
		}
		cout << ans;
		pq.pop();
	}
	return 0;
}
```
